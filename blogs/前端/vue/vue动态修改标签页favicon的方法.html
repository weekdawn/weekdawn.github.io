    vue动态修改标签页favicon的方法

    Week.D.Awn 4/12/2023 vueicon

    # vue动态修改标签页favicon的方法

    在项目的main.js中添加即可实现动态修改标签页favicon

        //修改Favicon的方法
    const changeFavicon = link => {
      let $favicon = document.querySelector('link[rel="icon"]');
      if ($favicon !== null) {
        $favicon.href = link;
      } else {
        $favicon = document.querySelector("link[rel*='icon']") || document.createElement("link");
        // 设置 link 元素的属性
        $favicon.type = 'image/x-icon';
        $favicon.rel = "icon";
        $favicon.href = link;
        document.head.appendChild($favicon);
      }
    };
    //根据传递的参数修改Favicon
    const changeIco = (type) => {
      // 得到图标地址
      let iconUrl
      if (type=='bd') {
        iconUrl = `./faviconbd.ico`
      } else {
        iconUrl = `./favicon.ico`
      }
      changeFavicon(iconUrl);
    };
    changeIco('bd');
