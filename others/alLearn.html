    const res = [];
    const traverse = (node: TreeNode) => {
        if(node === null) return;
        res.push(node.val)
        traverse(node.left);
        traverse(node.right);
    }
    traverse(root)
    return res;
};

    function preorderTraversal(root: TreeNode | null): number[] {
    // 递归终止条件：空节点返回空数组
    if (root === null) return [];
    const left = preorderTraversal(root.left);  // 递归左子树
    const right = preorderTraversal(root.right); // 递归右子树
    // 合并结果：根 → 左 → 右
    return [root.val, ...left, ...right];
}

    # 快排的分治写法：

    function quickSort(arr) {
    if (arr.length <= 1) return arr; // 基线条件
    const pivot = arr[0]; // 选择基准值（根）
    const left = [];
    const right = [];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] < pivot) left.push(arr[i]);  // 左子树（小于基准）
        else right.push(arr[i]);                // 右子树（大于等于基准）
    }
    // 合并：左 + 根 + 右
    return [...quickSort(left), pivot, ...quickSort(right)];
}

    # 判断链表中是否有环

    export function hasCycle(head: ListNode): boolean {
    // write code here
    if(!head || !head.next) return false;
    let slow = head;
    let fast = head.next;
    while(fast && fast.next) {
        slow = slow.next;
        fast = fast.next.next;
        if(slow === fast) return true;
    }
    return false;
}

    # 合并两个排序链表

    export function Merge(pHead1: ListNode, pHead2: ListNode): ListNode {
    // write code here
        // 创建一个虚拟头节点，简化操作
    let dummy = new ListNode(-1);
    let current = dummy;
    
    // 遍历两个链表
    while (pHead1 !== null && pHead2 !== null) {
        if (pHead1.val <= pHead2.val) {
            current.next = pHead1;
            pHead1 = pHead1.next;
        } else {
            current.next = pHead2;
            pHead2 = pHead2.next;
        }
        current = current.next;
    }
    
    // 将剩余链表直接连接到新链表末尾
    if (pHead1 !== null) {
        current.next = pHead1;
    } else {
        current.next = pHead2;
    }
    
    return dummy.next;
}

    # 大数相加

    export function solve(s: string, t: string): string {
    // write code here
    let i = s.length - 1;
    let j = t.length - 1;
    let carry = 0;
    let result = [];
    
    while (i >= 0 || j >= 0 || carry > 0) {
        const digitS = i >= 0 ? parseInt(s[i]) : 0;
        const digitT = j >= 0 ? parseInt(t[j]) : 0;
        
        const sum = digitS + digitT + carry;
        result.unshift(sum % 10);
        carry = Math.floor(sum / 10);
        
        i--;
        j--;
    }
    
    return result.join('');
}

    export function mergeTrees(t1: TreeNode, t2: TreeNode): TreeNode {
    // write code here
    if(!t1) return t2;
    if(!t2) return t1;
    let head = new TreeNode(t1.val + t2.val);
    head.left = mergeTrees(t1.left, t2.left);
    head.right = mergeTrees(t1.right, t2.right);
    return head;
}

    # 判断是不是平衡二叉树

    export function IsBalanced_Solution(pRoot: TreeNode): boolean {
    // write code here
    const checkBalance = (node) => {
        // 递归基：空节点高度为0
        if (!node) return 0;
        // 检查左子树
        const left = checkBalance(node.left);
        if (left === false) return false; // 左子树不平衡
        // 检查右子树
        const right = checkBalance(node.right);
        if (right === false) return false; // 右子树不平衡
        // 判断当前子树是否平衡
        return Math.abs(left - right) <= 1 
            ? Math.max(left, right) + 1  // 平衡则返回高度
            : false;                     // 不平衡返回false
    }
    return checkBalance(pRoot) !== false;
}

    # 数组中只出现一次的两个数字

    export function FindNumsAppearOnce(nums: number[]): number[] {
    // write code here
    let res = []
    let map = new Map<number, number>()
    nums.forEach((i) => {
        if(map.has(i)) {
            map.set(i, map.get(i) + 1)
        } else {
            map.set(i, 1)
        }
    })
    map.forEach((v,k) => {
        if(v===1) {
            res.push(k)
        }
    })
    return res.sort((a,b) => a-b);
}

    # 最长公共前缀

    export function longestCommonPrefix(strs: string[]): string {
    // write code here
    if (strs.length === 0) return "";
    strs.sort();
    const first = strs[0], last = strs[strs.length - 1];
    let i = 0;
    while (i < first.length && i < last.length && first[i] === last[i]) {
        i++;
    }
    return first.substring(0, i);
}

    export function isValid(s: string): boolean {
    // write code here
    if (s.length & 1) return false;
    let len = s.length;
    for (let i = 0; i < len / 2; i++) {
        s = s.replace("{}", "").replace("()", "").replace("[]", "");
    }
    return s === "";
}

    # 数组中出现次数超过一半的数字

    export function MoreThanHalfNum_Solution(numbers: number[]): number {
    // write code here
    if(numbers.length <= 1) return numbers[0];
    const map = new Map();
    map.set(numbers[0], 0);
    for(let n of numbers) {
        if(map.has(n)) {
            map.set(n, map.get(n) + 1);
            if(map.get(n) >= numbers.length/2) return n;
        } else {
            map.set(n, 1);
        }
    }
}

    # 没有重复项数字的全排列

    export function permute(num: number[]): number[][] {
    // write code here
    const result = [];
    num.sort((a, b) => a - b); // 先排序

    function backtrack(current, remaining) {
        if (remaining.length === 0) {
            result.push([...current]);
            return;
        }

        for (let i = 0; i < remaining.length; i++) {
            // 选择当前元素
            current.push(remaining[i]);
            // 移除已选元素，生成新数组
            const newRemaining = remaining
                .slice(0, i)
                .concat(remaining.slice(i + 1));
            // 递归
            backtrack(current, newRemaining);
            // 回溯
            current.pop();
        }
    }

    backtrack([], num);
    return result;
}

    # 最小花费爬楼梯

    export function minCostClimbingStairs(cost: number[]): number {
    // write code here
    const dp =[];
    dp[0] = 0;
    dp[1] = 0;
    for(let i = 2; i<=cost.length; i++) {
        dp[i] = Math.min(dp[i-1] + cost[i-1], dp[i-2] + cost[i-2]);
    }
    return dp[cost.length];
}

    # 跳台阶

    export function jumpFloor(number: number): number {
    // write code here
    const dp = [];
    dp[1] = 1;
    dp[2] = 2;
    for(let i = 3; i <= number; i++) {
        dp[i] = dp[i-1] + dp[i-2]
    }
    return dp[number];
}
